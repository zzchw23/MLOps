def feateng3(df):
    data = {'eat': [
        'I eat whatever is convenient|',
        'I eat strict Paleo|',
        "I eat quality foods but don't measure the amount|I eat 1-3 full cheat meals per week|",
        'I weigh and measure my food|I eat strict Paleo|',
        "I eat strict Paleo|I eat quality foods but don't measure the amount|I eat whatever is convenient|",
        'I weigh and measure my food|',
        'Decline to answer|', np.nan]}
    data = pd.DataFrame(data)

    data['Paleo'] = data['eat'].str.contains('I eat strict Paleo', na=False).astype(int)
    data['Measures'] = data['eat'].str.contains('I weigh and measure my food', na=False).astype(int)
    non_paleo_pattern = (
        "I eat whatever is convenient|"
        "I eat 1-3 full cheat meals per week|"
        "I eat quality foods but don't measure the amount")
    data['Non-paleo'] = data['eat'].str.contains(non_paleo_pattern, na=False).astype(int)
    data = data.drop(['eat'], axis=1)
    df = pd.concat([df, data], axis=1)
    df = df.drop(['eat'], axis=1)
    return df
df = feateng3(df)

def feateng4(df):
    data = {'country': [
    'Southern California', 'Africa', 'North East', 'Australia',
    'Latin America', 'Europe', 'Canada West', 'Asia',
    'South East', 'South Central', 'North West', 'North Central',

        'Canada': 'Canada West|Canada East',
        'Africa': 'Africa',
        'Asia': 'Asia',
        'LatinAmerica': 'Latin America',
        'Europe': 'Europe',
        'Australia': 'Australia'}
    new_columns = list(categories.keys())
    for col_name in new_columns:
        df[col_name] = 0
    for col_name, pattern in categories.items():
        df[col_name] = np.where(df['region'].str.contains(pattern, na=False), 1, 0)
    df = df.drop(['region'], axis=1)
    return df

numeric_features = X.select_dtypes(include=np.number).columns.tolist()
categorical_features = X.select_dtypes(include='category').columns.tolist()

numeric_preprocess = Pipeline(steps=[
    ("imputer", SimpleImputer(strategy="median")),
    ("scaler", StandardScaler()),
    # Optional: include polynomial interactions to show feature generation
    ("poly", PolynomialFeatures(degree=2, include_bias=False))
])

categorical_preprocess = Pipeline(steps=[
    ("imputer", SimpleImputer(strategy="most_frequent")),
    ("onehot", OneHotEncoder(handle_unknown="ignore"))
])

preprocessor = ColumnTransformer(transformers=[
    ("num", numeric_preprocess, numeric_features),
    ("cat", categorical_preprocess, categorical_features)
])

clf = LogisticRegression(max_iter=200)

pipe = Pipeline(steps=[
    ("preprocess", preprocessor),
